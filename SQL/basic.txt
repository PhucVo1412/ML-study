C1:
CREATE TABLE Salespersons(
    SalespersonsID INT PRIMARY KEY,
    FullName VARCHAR(100),
    Email VARCHAR(100) UNIQUE,
    PhoneNumber VARCHAR(10)
);

C2:
CREATE TABLE Customers(
    CustomersID INT PRIMARY KEY,
    FullName VARCHAR(100),
    Email VARCHAR(100) UNIQUE,
    PhoneNumber VARCHAR(10)
);

C3:
CREATE TABLE Orders(
    OrderID INT PRIMARY KEY,
    CustomerID INT,
    SalespersonID INT,
    OrderDate DATE,
    TotalAmount DECIMAL,

    FOREIGN KEY (CustomerID) REFERENCES Customers(CustomersID),
    FOREIGN KEY (SalespersonID) REFERENCES Salespersons(SalespersonsID)
);

C8:
SELECT Customers.FullName, COUNT(Orders.OrderID) AS TotalOrders
FROM Orders
JOIN Customers ON Orders.CustomerID = Customers.CustomersID
GROUP BY Customers.FullName
ORDER BY TotalOrders DESC
LIMIT 1;

SELECT Salespersons.FullName, SUM(Orders.TotalAmount) AS TotalSales
FROM Orders
JOIN Salespersons ON Orders.SalespersonID = Salespersons.SalespersonsID
GROUP BY Salespersons.FullName;

C10:
